Dokumentasi

Mudah Diintegrasikan

Mulai kirim pesan dalam hitungan menit dengan dokumentasi yang jelas

send_message.php 
<?php
$curl = curl_init();

curl_setopt_array($curl, array(
  CURLOPT_URL => 'http://localhost:5000/api/send',
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_ENCODING => '',
  CURLOPT_MAXREDIRS => 10,
  CURLOPT_TIMEOUT => 0,
  CURLOPT_FOLLOWLOCATION => true,
  CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
  CURLOPT_CUSTOMREQUEST => 'POST',
  CURLOPT_POSTFIELDS => json_encode(array(
    'recipient' => '6281234567890',
    'message' => 'Halo dari WA Gateway!'
  )),
  CURLOPT_HTTPHEADER => array(
    'X-API-Key: YOUR_DEVICE_API_TOKEN',
    'Content-Type: application/json'
  ),
));

$response = curl_exec($curl);
curl_close($curl);

echo $response;
?>
sendMessage.js 
const axios = require('axios');

const sendMessage = async () => {
  try {
    const response = await axios.post(
      'https://api.wagateway.com/send',
      {
        target: '6281234567890',
        message: 'Halo dari WA Gateway!'
      },
      {
        headers: {
          'Authorization': 'YOUR_API_TOKEN'
        }
      }
    );
    
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
};

sendMessage();
send_message.py 
import requests
import json

url = "http://localhost:5000/api/send"

payload = {
    'recipient': '6281234567890',
    'message': 'Halo dari WA Gateway!'
}
headers = {
    'X-API-Key': 'YOUR_DEVICE_API_TOKEN',
    'Content-Type': 'application/json'
}

response = requests.post(url, json=payload, headers=headers)

print(response.text)
